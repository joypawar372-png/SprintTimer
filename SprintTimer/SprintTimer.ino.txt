bool isLaserAligned() {
    return (digitalRead(LASER_SENSOR_PIN) == LASER_ALIGNED_STATE);
}

void triggerSimpleBeep(int frequency, int durationMs) {
    tone(BUZZER_PIN, frequency, durationMs);
    delay(durationMs);
    noTone(BUZZER_PIN);
}

int getBatteryPercentage() {
    long sum = 0;
    for(int i = 0; i < 10; i++) {
        sum += analogRead(BATTERY_ADC_PIN);
        delayMicroseconds(50);
    }
    float avgRaw = sum / 10.0;
    float calculatedVoltage = (avgRaw / 4095.0) * 3.3 * 2.0; 
    int capacityPercent = (int)((calculatedVoltage - 3.4) / (4.2 - 3.4) * 100.0);
    return constrain(capacityPercent, 0, 100);
}

void turnWifiOn() {
    WiFi.mode(WIFI_AP);
    WiFi.softAP(ssid, password);
    server.begin();
    isWifiActive = true;
}

void turnWifiOff() {
    server.stop();
    WiFi.softAPdisconnect(true);
    WiFi.mode(WIFI_OFF);
    isWifiActive = false;
}

void showStandbyScreen() {

    drawBatteryOverlay();

    display.setCursor(8, 22);
    if (currentState == IDLE) {
        display.println("STATUS: UNARMED");
        display.setCursor(8, 42);

void executeSystemSleep() {
    display.clearDisplay();
    display.display();
    display.ssd1306_command(SSD1306_DISPLAYOFF); 
    digitalWrite(RED_LED_PIN, LOW);
    digitalWrite(GREEN_LED_PIN, LOW);
    
    bool cachedRadioState = isWifiActive;
    if (isWifiActive) turnWifiOff();
    
    esp_sleep_enable_ext0_wakeup((gpio_num_t)START_BUTTON_PIN, HIGH);
    esp_light_sleep_start();
    
    display.ssd1306_command(SSD1306_DISPLAYON);
    if (cachedRadioState) turnWifiOn();
    
    lastSystemActivityMillis = millis();
    currentState = IDLE;
    triggerSimpleBeep(2000, 150);
    if(isLaserAligned()) showStandbyScreen(); else showLaserAlignmentWarning();
}

void startSpecialModeWithCountdown() {
    runIdCounter++; 
    for (int i = 3; i > 0; i--) {
        display.clearDisplay();
        display.setTextColor(SSD1306_WHITE);
        display.setTextSize(1); display.setCursor(4, 4);
        display.println("SPECIAL MODE TIMER");
        display.drawFastHLine(0, 14, 128, SSD1306_WHITE);
        display.setCursor(40, 30);
        display.print("COUNT:  "); display.print(i);
        display.display();
        
        tone(BUZZER_PIN, 1200, 150);
        
        unsigned long startCountdownBracketMillis = millis();
        bool alternateFlashState = false;
        while(millis() - startCountdownBracketMillis < 1000) {
            alternateFlashState = !alternateFlashState;
            digitalWrite(RED_LED_PIN, alternateFlashState ? HIGH : LOW);
            digitalWrite(GREEN_LED_PIN, alternateFlashState ? LOW : HIGH);
            delay(125); 
        }
    }
    
    digitalWrite(RED_LED_PIN, LOW);
    digitalWrite(GREEN_LED_PIN, HIGH); 
    
    display.clearDisplay();
    display.setTextSize(1); display.setCursor(4, 4);
    display.println("SPECIAL MODE TIMER");
    display.drawFastHLine(0, 14, 128, SSD1306_WHITE);
    display.setCursor(40, 30);
    display.print("LAUNCH GO!");
    display.display();
    triggerSimpleBeep(2200, 400);
    
    startTimeMicros = micros();
    lastPassMicros = startTimeMicros;
    currentState = RUNNING;
    currentLapSequenceNumber = 1;
}

        function drawLocalChart(laps) {
            if(laps.length < 2) return '';
            let times = laps.map(l => l.time);
            let min = Math.min(...times), max = Math.max(...times);
            let range = max - min || 1;
            let w = 500, h = 60;
            let pad = 10;
            let step = (w - pad*2) / (laps.length - 1);
            let pts = "";
            laps.forEach((l, i) => {
                let x = pad + (i * step);
                let y = h - pad - (((l.time - min) / range) * (h - pad*2));
                pts += `${x},${y} `;
            });
            return `<svg viewBox="0 0 ${w} ${h}" class="svg-box"><polyline fill="none" stroke="#00ffcc" stroke-width="2" points="${pts.trim()}"/></svg>`;

void handleRoot() { server.send(200, "text/html", HTML_PAGE); }
void handleWatchView() { server.send(200, "text/html", WATCH_PAGE); }

void handleArmCommand() {
    lastSystemActivityMillis = millis();
    if (currentState == OTA_UPDATE) {
        server.send(200, "text/plain", "SYSTEM_FLASH_LOCK");
        return;
    }
    
    if (currentState == IDLE || currentState == DISPLAY_SCORE || currentState == FINISHED) {
        if (!isLaserAligned()) {
            server.send(200, "text/plain", "FAULT_MISALIGNED");
            return;
        }
        
        if (activeMode == MODE_SPECIAL) {
            server.send(200, "text/plain", "COUNTDOWN_STARTED");
            startSpecialModeWithCountdown(); 
        } else {
            currentState = ARM_WAIT_START;
            laserWasAlignedSinceArming = false; 
            currentLapSequenceNumber = 0;
            server.send(200, "text/plain", "ARMED_READY");
        }
    } else {
        server.send(200, "text/plain", "SYSTEM_BUSY");
    }
}

void handleSetModeCommand() {
    lastSystemActivityMillis = millis();
    if (currentState == OTA_UPDATE) return;
    String target = server.arg("mode");
    activeMode = (target == "special") ? MODE_SPECIAL : MODE_MAIN;
    currentState = IDLE;
    server.send(200, "text/plain", "MODE_CHANGED");
}

void handleStatus() {
    String stateStr = "IDLE";
    if (currentState == ARM_WAIT_START) stateStr = "READY_ARMED";
    else if (currentState == RUNNING) stateStr = "RUNNING";
    else if (currentState == FINISHED || currentState == DISPLAY_SCORE) stateStr = "FINISHED";
    else if (currentState == OTA_UPDATE) stateStr = "FLASHING_OTA";

    String modeStr = (activeMode == MODE_MAIN) ? "MODE_MAIN" : "MODE_SPECIAL";
    String laserStr = isLaserAligned() ? "ALIGNED" : "MISALIGNED";
    
    String json = "{\"state\":\"" + stateStr + "\",\"mode\":\"" + modeStr + "\",\"laser\":\"" + laserStr + 
                  "\",\"time\":" + String(finalTimeSeconds, 3) + ",\"lapIndex\":" + String(currentLapSequenceNumber) + 
                  ",\"battery\":" + String(getBatteryPercentage()) + ",\"runId\":" + String(runIdCounter) + "}";
    server.send(200, "application/json", json);
}

void executeButtonCommand(int actionType) {
    if (currentState == OTA_UPDATE) return; 
    lastSystemActivityMillis = millis(); 

    if (actionType == 6) { 
        if (isWifiActive) turnWifiOff(); else turnWifiOn();
        triggerSimpleBeep(1800, 100); delay(50); triggerSimpleBeep(1800, 100);
        if (currentState == INFO_SCREEN) showInfoScreen();
        else if (currentState == IDLE) {
            if(isLaserAligned()) showStandbyScreen(); else showLaserAlignmentWarning();
        }
        return;
    }

    if (currentState == IDLE && actionType == 2) { 
        activeMode = (activeMode == MODE_SPECIAL) ? MODE_MAIN : MODE_SPECIAL;
        triggerSimpleBeep(1500, 100);
        if(isLaserAligned()) showStandbyScreen(); else showLaserAlignmentWarning();
        return;
    }

    if (currentState == RUNNING && actionType == 1) { 
        currentState = FINISHED;
        return;
    }

    if (currentState == DISPLAY_SCORE || currentState == FINISHED || currentState == INFO_SCREEN) {
        if (actionType == 1) { 
            finalTimeSeconds = 0.0; 
            currentState = IDLE;
            currentLapSequenceNumber = 0;
            digitalWrite(GREEN_LED_PIN, LOW); 
            digitalWrite(RED_LED_PIN, LOW);
            if(isLaserAligned()) showStandbyScreen(); else showLaserAlignmentWarning();
        }
        return; 
    }

    if (actionType == 3) { 
        if (currentState == IDLE) {
            if (!isLaserAligned()) {
                triggerSimpleBeep(600, 300);
                showLaserAlignmentWarning();
                return;
            }
            
            if (activeMode == MODE_SPECIAL) {
                startSpecialModeWithCountdown();
            } else {
                currentState = ARM_WAIT_START;
                laserWasAlignedSinceArming = false; 
                currentLapSequenceNumber = 0;
                triggerSimpleBeep(2000, 250);
                showStandbyScreen();
            }
        }
    } 
    else if (actionType == 4) { 
        if (currentState == IDLE) {
            currentState = INFO_SCREEN;
            showInfoScreen();
        }
    }
}

void checkStartButtonEvents() {
    int reading = digitalRead(START_BUTTON_PIN);
    static int lastReading = LOW; 
    static unsigned long pressedTime = 0;
    static unsigned long releasedTime = 0;
    static bool isPressing = false;
    static bool longPressedActive = false;
    static int clickCounter = 0;

    if (reading == HIGH && lastReading == LOW) { 
        pressedTime = millis();
        isPressing = true;
        longPressedActive = false;
    }
    
    if (reading == HIGH && isPressing) {
        if (!longPressedActive && (millis() - pressedTime > 2000)) { 
            longPressedActive = true;
            isPressing = false;
            clickCounter = 0;
            executeButtonCommand(3); 
        }
    }

    if (reading == LOW && lastReading == HIGH) { 
        if (!longPressedActive) {
            clickCounter++;
            releasedTime = millis();
        }
        isPressing = false;
    }

    if (reading == LOW && clickCounter > 0 && (millis() - releasedTime > 280)) {
        if (clickCounter == 1) executeButtonCommand(1);      
        else if (clickCounter == 2) executeButtonCommand(2); 
        else if (clickCounter == 3) executeButtonCommand(4); 
        else if (clickCounter == 6) executeButtonCommand(6); 
        clickCounter = 0;
    }
    lastReading = reading;
}

void setup() {
    WRITE_PERI_REG(RTC_CNTL_BROWN_OUT_REG, 0); 
    WiFi.mode(WIFI_OFF); 
    
    pinMode(START_BUTTON_PIN, INPUT); 
    pinMode(LASER_SENSOR_PIN, INPUT_PULLUP);
    analogSetAttenuation(ADC_11db);
    
    pinMode(RED_LED_PIN, OUTPUT);
    pinMode(GREEN_LED_PIN, OUTPUT);
    pinMode(BUZZER_PIN, OUTPUT); 
    
    digitalWrite(RED_LED_PIN, LOW);
    digitalWrite(GREEN_LED_PIN, LOW);
    digitalWrite(BUZZER_PIN, LOW);

    if(!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
        while(true) { 
            digitalWrite(RED_LED_PIN, HIGH); delay(100);
            digitalWrite(RED_LED_PIN, LOW); delay(100);
        }
    }

    display.clearDisplay();
    display.setTextColor(SSD1306_WHITE);
    display.setTextSize(1);
    display.setCursor(16, 28);
    display.println("SYSTEM REBOOT");
    display.display();
    
    triggerSimpleBeep(1500, 200);
    delay(200);
    
    server.on("/", HTTP_GET, handleRoot);
    server.on("/watch", HTTP_GET, handleWatchView); 
    server.on("/arm", HTTP_GET, handleArmCommand);
    server.on("/status", HTTP_GET, handleStatus);
    server.on("/setMode", HTTP_GET, handleSetModeCommand);
    
    // ====================================================
    // HIGH-STABILITY MULTI-PART PHONE UPLOAD HANDLING BLOCK
    // ====================================================
    server.on("/update", HTTP_POST, []() {
        server.sendHeader("Connection", "close");
        server.send(200, "text/plain", (Update.hasError()) ? "OTA_ERROR" : "OTA_SUCCESS_REBOOTING");
        delay(1000);
        ESP.restart();
    }, []() {
        HTTPUpload& upload = server.upload();
        if (upload.status == UPLOAD_FILE_START) {
            currentState = OTA_UPDATE;
            digitalWrite(RED_LED_PIN, HIGH);
            digitalWrite(GREEN_LED_PIN, HIGH);
            
            display.clearDisplay();
            display.setTextSize(1); display.setCursor(4, 4);
            display.println("FIRMWARE UPDATE");
            display.drawFastHLine(0, 14, 128, SSD1306_WHITE);
            display.setCursor(4, 28);
            display.println("PHONE STREAM ACTIVE");
            display.setCursor(4, 45);
            display.println("Flashing Core...");
            display.display();
            
            // Read exact byte allocation sent by mobile JS client
            size_t absoluteBinarySize = UPDATE_SIZE_UNKNOWN;
            if (server.hasArg("size")) {
                absoluteBinarySize = server.arg("size").toInt();
            }
            
            if (!Update.begin(absoluteBinarySize)) {
                Update.printError(Serial);
            }
        } else if (upload.status == UPLOAD_FILE_WRITE) {
            if (Update.write(upload.buf, upload.currentSize) != upload.currentSize) {
                Update.printError(Serial);
            }
            
            // ─── THE ANTI-DISCONNECT FIX ───
            // Feed the RTOS watchdogs and let the TCP stack transmit ACK packets back to the phone
            yield();
            delay(2); 
            
        } else if (upload.status == UPLOAD_FILE_END) {
            if (Update.end(true)) {
                display.clearDisplay();
                display.setCursor(4, 20);
                display.println("PAYLOAD LOCKED");
                display.setCursor(4, 40);
                display.println("BOOTING TARGET...");
                display.display();
                triggerSimpleBeep(2500, 350);
            }
        }
    });
    
    turnWifiOn();
    lastSystemActivityMillis = millis();
    
    if(isLaserAligned()) showStandbyScreen(); else showLaserAlignmentWarning();
}

// ==========================================
// CYCLIC REAL-TIME CORE MONITOR ENGINE
// ==========================================
void loop() {
    if (isWifiActive) {
        server.handleClient();
    }
    
    if (currentState == OTA_UPDATE) return; 
    
    checkStartButtonEvents();
    unsigned long currentMillis = millis();

    if (currentState == IDLE && (currentMillis - lastSystemActivityMillis >= INACTIVITY_TIMEOUT_MILLIS)) {
        executeSystemSleep();
        return; 
    }

    if (currentState == IDLE || currentState == ARM_WAIT_START) {
        if (!isLaserAligned()) { 
            if (currentMillis - ledBlinkTimerMillis >= 300) {
                ledBlinkTimerMillis = currentMillis;
                digitalWrite(RED_LED_PIN, !digitalRead(RED_LED_PIN));
            }
            digitalWrite(GREEN_LED_PIN, LOW);
            if (currentMillis - menuRefreshTimerMillis >= 1000) {
                menuRefreshTimerMillis = currentMillis;
                showLaserAlignmentWarning();
            }
        } 
        else {
            digitalWrite(RED_LED_PIN, LOW);
            digitalWrite(GREEN_LED_PIN, currentState == ARM_WAIT_START ? !digitalRead(GREEN_LED_PIN) : HIGH);
            if (currentMillis - menuRefreshTimerMillis >= 1000) {
                menuRefreshTimerMillis = currentMillis;
                showStandbyScreen();
            }
        }
    }

    if (currentState == ARM_WAIT_START && activeMode == MODE_MAIN) {
        if (isLaserAligned()) {
            laserWasAlignedSinceArming = true;
        }
        
        if (!isLaserAligned()) {
            if (laserWasAlignedSinceArming) {
                lastSystemActivityMillis = currentMillis;
                startTimeMicros = micros();
                lastPassMicros = startTimeMicros;
                currentState = RUNNING;
                currentLapSequenceNumber = 1;
                runIdCounter++; 
                
                digitalWrite(RED_LED_PIN, HIGH);
                digitalWrite(GREEN_LED_PIN, LOW);
                triggerSimpleBeep(1000, 80);
                
                display.clearDisplay();
                display.setTextSize(1); display.setCursor(4,4); display.print("TRACK ACTIVE");
                display.setCursor(4, 24); display.print("LAP 1 IN PROGRESS...");
                display.display();
                
                delayMicroseconds(PASS_LOCKOUT_MICROS); 
            } else {
                currentState = IDLE;
                triggerSimpleBeep(700, 300); 
                showLaserAlignmentWarning();
            }
        }
    }

    if (currentState == RUNNING && !isLaserAligned()) {
        unsigned long currentMicros = micros();
        
        if (activeMode == MODE_SPECIAL && (currentMicros - startTimeMicros < SPECIAL_DEADZONE_MICROS)) {
            // Ignition protection blanking window loop
        } 
        else if (currentMicros - lastPassMicros >= PASS_LOCKOUT_MICROS) {
            lastSystemActivityMillis = currentMillis;
            finalTimeSeconds = (currentMicros - lastPassMicros) / 1000000.0;
            lastPassMicros = currentMicros;
            
            triggerSimpleBeep(1500, 80);
            
            display.clearDisplay();
            display.setTextSize(1); display.setCursor(4,4); display.print("SPLIT LOGGED");
            display.setCursor(4, 24);
            display.print("LAP "); display.print(currentLapSequenceNumber); display.print(": "); display.print(finalTimeSeconds, 3); display.print("s");
            display.display();
            
            currentLapSequenceNumber++;
            delayMicroseconds(PASS_LOCKOUT_MICROS);
        }
    }

    if (currentState == FINISHED) {
        lastSystemActivityMillis = currentMillis;
        
        display.clearDisplay();
        display.setTextSize(1); display.setCursor(4, 15); display.println("RUN COMPLETE");
        display.setCursor(4, 35); 
        display.print("TOTAL LAPS: "); display.print(currentLapSequenceNumber - 1);
        display.display();

        for (int i = 0; i < 3; i++) {
            digitalWrite(RED_LED_PIN, HIGH); digitalWrite(GREEN_LED_PIN, HIGH);
            triggerSimpleBeep(1600, 60); delay(40);             
            digitalWrite(RED_LED_PIN, LOW); digitalWrite(GREEN_LED_PIN, LOW);
            delay(40);              
        }
        digitalWrite(GREEN_LED_PIN, HIGH); 
        currentState = DISPLAY_SCORE;
    }
}
